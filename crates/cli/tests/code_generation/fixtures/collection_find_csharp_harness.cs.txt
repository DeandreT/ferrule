using Ferrule.Generated;
using Ferrule.Runtime;

var catalog = CatalogSource();
var inputs = new NamedInput[] { new("catalog", catalog) };

AssertInstance(
    GeneratedMapping.ExecuteWithSources(PrimarySource(false), inputs),
    Expected(),
    "normal");
AssertError(
    FerruleRuntimeError.NotABool,
    61U,
    () => GeneratedMapping.ExecuteWithSources(PrimarySource(true), inputs));
AssertError(
    FerruleRuntimeError.MissingSourceField,
    null,
    () => GeneratedMapping.ExecuteWithSources(PrimaryWithoutDepartments(), inputs));

static FerruleGroup PrimarySource(bool failNonBool) => new(new FerruleField[]
{
    new("Needle", Scalar(FerruleValue.FromString("Ada"))),
    new("Suffix", Scalar(FerruleValue.FromString("!"))),
    new("FailNonBool", Scalar(FerruleValue.FromBoolean(failNonBool))),
    new("Departments", new FerruleRepeated(new FerruleInstance[]
    {
        Department(
            "Remote",
            new FerruleInstance[]
            {
                Person("Ada", "Wrong: ", "remote@example.test", FerruleValue.Null),
                Person("Lin", "Lead: ", "lin@example.test", FerruleValue.XmlNil),
            }),
        Department(
            "HQ",
            new FerruleInstance[]
            {
                Person(
                    "Grace",
                    "Director: ",
                    "grace@example.test",
                    FerruleValue.FromBoolean(false)),
                Person(
                    "Ada",
                    "Engineer: ",
                    "ada@example.test",
                    FerruleValue.FromBoolean(true)),
            }),
    })),
});

static FerruleGroup PrimaryWithoutDepartments() => new(new FerruleField[]
{
    new("Needle", Scalar(FerruleValue.FromString("Ada"))),
    new("Suffix", Scalar(FerruleValue.FromString("!"))),
    new("FailNonBool", Scalar(FerruleValue.FromBoolean(false))),
});

static FerruleGroup Department(string office, IEnumerable<FerruleInstance> people) =>
    new(new FerruleField[]
    {
        new("Office", Scalar(FerruleValue.FromString(office))),
        new("People", new FerruleRepeated(people)),
    });

static FerruleGroup Person(
    string first,
    string title,
    string email,
    FerruleValue decision) => new(new FerruleField[]
{
    new("First", Scalar(FerruleValue.FromString(first))),
    new("Title", Scalar(FerruleValue.FromString(title))),
    new("Email", Scalar(FerruleValue.FromString(email))),
    new("Decision", Scalar(decision)),
});

static FerruleGroup CatalogSource() => new(new FerruleField[]
{
    new("Items", new FerruleRepeated(new FerruleInstance[]
    {
        CatalogItem("Ada", "catalog-A"),
        CatalogItem("Ada", "catalog-second"),
        CatalogItem("Grace", "catalog-G"),
    })),
});

static FerruleGroup CatalogItem(string key, string label) => new(new FerruleField[]
{
    new("Key", Scalar(FerruleValue.FromString(key))),
    new("Label", Scalar(FerruleValue.FromString(label))),
});

static FerruleGroup Expected() => new(new FerruleField[]
{
    new("Details", Scalar(FerruleValue.FromString("Engineer: ada@example.test!"))),
    new("DepartmentPosition", Scalar(FerruleValue.FromInt64(2))),
    new("PersonPosition", Scalar(FerruleValue.FromInt64(2))),
    new("CatalogValue", Scalar(FerruleValue.FromString("catalog-A!"))),
    new("NullableDecision", Scalar(FerruleValue.FromString("ada@example.test"))),
    new("LazyMiss", Scalar(FerruleValue.Null)),
    new("NonBoolProbe", Scalar(FerruleValue.FromString("safe"))),
});

static FerruleScalar Scalar(FerruleValue value) => new(value);

static void AssertError(FerruleRuntimeError expected, uint? node, Action action)
{
    try
    {
        action();
    }
    catch (FerruleRuntimeException error) when (
        error.Error == expected && error.Node == node)
    {
        return;
    }
    throw new InvalidOperationException($"Expected {expected} at node {node}.");
}

static void AssertInstance(FerruleInstance actual, FerruleInstance expected, string label)
{
    switch (actual, expected)
    {
        case (FerruleScalar left, FerruleScalar right) when left.Value == right.Value:
            return;
        case (FerruleGroup left, FerruleGroup right) when left.Fields.Count == right.Fields.Count:
            for (var index = 0; index < left.Fields.Count; index++)
            {
                if (left.Fields[index].Name != right.Fields[index].Name)
                {
                    break;
                }
                AssertInstance(left.Fields[index].Value, right.Fields[index].Value, label);
                if (index + 1 == left.Fields.Count)
                {
                    return;
                }
            }
            break;
        case (FerruleRepeated left, FerruleRepeated right) when left.Items.Count == right.Items.Count:
            for (var index = 0; index < left.Items.Count; index++)
            {
                AssertInstance(left.Items[index], right.Items[index], label);
            }
            return;
    }

    throw new InvalidOperationException($"Generated {label} output differs from the engine.");
}
