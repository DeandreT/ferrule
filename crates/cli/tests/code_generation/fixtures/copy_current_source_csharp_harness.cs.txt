using Ferrule.Generated;
using Ferrule.Runtime;

var source = CopySource();
var output = (FerruleGroup)GeneratedMapping.Execute(source);

Assert(!ReferenceEquals(source, output));
Assert(output.Fields.Select(field => field.Name).SequenceEqual(
    new[] { "Id", "Missing", "Nil", "Empty", "Items" }));
Assert(((FerruleScalar)output.Fields[0].Value).Value == FerruleValue.FromInt64(7));
Assert(((FerruleScalar)output.Fields[1].Value).Value == FerruleValue.Null);
Assert(((FerruleScalar)output.Fields[2].Value).Value == FerruleValue.XmlNil);
Assert(((FerruleGroup)output.Fields[3].Value).Fields.Count == 0);
var items = (FerruleMappedSequence)output.Fields[4].Value;
Assert(items.Items.Count == 2);
Assert(!ReferenceEquals(source.Fields[4].Value, items));
Assert(ItemName(items.Items[0]) == "first");
Assert(ItemName(items.Items[1]) == "second");

var error = Error(() => GeneratedMapping.Execute(
    new FerruleScalar(FerruleValue.FromString("wrong"))));
Assert(error.Error == FerruleRuntimeError.CopyCurrentSourceRequiresGroup);
Assert(error.Detail == "scalar");

static FerruleGroup CopySource() => new(new FerruleField[]
{
    new("Id", new FerruleScalar(FerruleValue.FromInt64(7))),
    new("Missing", new FerruleScalar(FerruleValue.Null)),
    new("Nil", new FerruleScalar(FerruleValue.XmlNil)),
    new("Empty", new FerruleGroup(Array.Empty<FerruleField>())),
    new(
        "Items",
        new FerruleMappedSequence(new FerruleInstance[]
        {
            new FerruleGroup(new FerruleField[]
            {
                new("Name", new FerruleScalar(FerruleValue.FromString("first"))),
            }),
            new FerruleGroup(new FerruleField[]
            {
                new("Name", new FerruleScalar(FerruleValue.FromString("second"))),
            }),
        })),
});

static string ItemName(FerruleInstance instance) =>
    ((FerruleScalar)((FerruleGroup)instance).Fields[0].Value).Value.StringValue;

static FerruleRuntimeException Error(Action action)
{
    try
    {
        action();
    }
    catch (FerruleRuntimeException exception)
    {
        return exception;
    }

    throw new InvalidOperationException("expected copy-current-source to fail");
}

static void Assert(bool condition)
{
    if (!condition)
    {
        throw new InvalidOperationException("copy-current-source output differs from the engine");
    }
}
