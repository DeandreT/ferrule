static FerruleGroup TargetRow(
    string label,
    string cValue,
    long cPosition,
    long aPosition,
    long joinPosition) => new(new FerruleField[]
{
    Field("ALabel", FerruleValue.FromString(label)),
    Field("BTag", FerruleValue.FromString("high")),
    Field("CValue", FerruleValue.FromString(cValue)),
    Field("JoinPosition", FerruleValue.FromInt64(joinPosition)),
    Field("APosition", FerruleValue.FromInt64(aPosition)),
    Field("BPosition", FerruleValue.FromInt64(2)),
    Field("CPosition", FerruleValue.FromInt64(cPosition)),
    Field("Rank", FerruleValue.FromInt64(30)),
    new("Details", Record(
        Field("Summary", FerruleValue.FromString($"{label}:{cValue}")))),
});

static FerruleGroup Record(params FerruleField[] fields) => new(fields);

static FerruleField Field(string name, FerruleValue value) => new(name, new FerruleScalar(value));

static void AssertInstance(FerruleInstance actual, FerruleInstance expected)
{
    switch (actual, expected)
    {
        case (FerruleScalar left, FerruleScalar right) when left.Value == right.Value:
            return;
        case (FerruleGroup left, FerruleGroup right) when left.Fields.Count == right.Fields.Count:
            for (var index = 0; index < left.Fields.Count; index++)
            {
                if (left.Fields[index].Name != right.Fields[index].Name)
                {
                    break;
                }
                AssertInstance(left.Fields[index].Value, right.Fields[index].Value);
                if (index + 1 == left.Fields.Count)
                {
                    return;
                }
            }
            break;
        case (FerruleRepeated left, FerruleRepeated right) when left.Items.Count == right.Items.Count:
            for (var index = 0; index < left.Items.Count; index++)
            {
                AssertInstance(left.Items[index], right.Items[index]);
            }
            return;
    }

    throw new InvalidOperationException("Generated join output differs from the engine.");
}
