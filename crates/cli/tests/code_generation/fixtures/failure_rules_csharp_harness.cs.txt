using Ferrule.Generated;
using Ferrule.Runtime;

var happy = Run(Case.None);
AssertInstance(happy.Primary, ExpectedOutput(), "primary");
if (happy.Extras.Count != 1 || happy.Extras[0].Name != "audit")
{
    throw new InvalidOperationException("Generated named output metadata differs from the engine.");
}
AssertInstance(happy.Extras[0].Instance, ExpectedOutput(), "audit");

Failure(Case.Source, 1, "invalid:B");
Failure(Case.Generated, 2, "2");
Failure(Case.Named, 3, "blocked");
Failure(Case.EmptyMessage, 4, string.Empty);

var primary = Primary(Case.NonBoolean);
var rules = Rules(Case.NonBoolean);
try
{
    GeneratedMapping.ExecuteWithSources(primary, new[] { new NamedInput("rules", rules) });
}
catch (FerruleRuntimeException error) when (
    error.Error == FerruleRuntimeError.NotABool && error.Node == 31U)
{
    return;
}
throw new InvalidOperationException("A non-boolean failure predicate did not retain its node.");

static ExecutionOutputs Run(Case value)
{
    var primary = Primary(value);
    var rules = Rules(value);
    return GeneratedMapping.ExecuteOutputsWithSources(
        primary,
        new[] { new NamedInput("rules", rules) });
}

static void Failure(Case value, int rule, string? message)
{
    try
    {
        Run(value);
    }
    catch (FerruleRuntimeException error) when (
        error.Error == FerruleRuntimeError.MappingFailure &&
        error.FailureRule == rule &&
        error.MappingFailureMessage == message)
    {
        return;
    }
    throw new InvalidOperationException($"Failure rule {rule} did not match the engine.");
}

static FerruleGroup Primary(Case value)
{
    var rows = value == Case.Source
        ? new[] { Row("A", true), Row("B", false), Row("C", false) }
        : new[] { Row("A", true), Row("B", true) };
    var empty = value == Case.EmptyMessage
        ? new FerruleInstance[] { new FerruleGroup(Array.Empty<FerruleField>()) }
        : Array.Empty<FerruleInstance>();
    var bad = value == Case.NonBoolean
        ? new FerruleInstance[] { Group("Value", FerruleValue.FromString("not-bool")) }
        : Array.Empty<FerruleInstance>();
    return new FerruleGroup(new FerruleField[]
    {
        new("Name", Scalar(FerruleValue.FromString("mapped"))),
        new("FailGenerated", Scalar(FerruleValue.FromBoolean(value == Case.Generated))),
        new("Row", new FerruleRepeated(rows)),
        new("EmptyTrigger", new FerruleRepeated(empty)),
        new("BadTrigger", new FerruleRepeated(bad)),
    });
}

static FerruleGroup Row(string code, bool valid) => new(new FerruleField[]
{
    new("Code", Scalar(FerruleValue.FromString(code))),
    new("Valid", Scalar(FerruleValue.FromBoolean(valid))),
});

static FerruleGroup Rules(Case value)
{
    var flag = (string code, bool reject) => new FerruleGroup(new FerruleField[]
    {
        new("Code", Scalar(FerruleValue.FromString(code))),
        new("Reject", Scalar(FerruleValue.FromBoolean(reject))),
    });
    return new FerruleGroup(new FerruleField[]
    {
        new("Flag", new FerruleRepeated(new FerruleInstance[]
        {
            flag("allowed", false),
            flag("blocked", value == Case.Named),
        })),
    });
}

static FerruleGroup ExpectedOutput() =>
    Group("Name", FerruleValue.FromString("mapped"));

static FerruleGroup Group(string name, FerruleValue value) =>
    new(new FerruleField[] { new(name, Scalar(value)) });

static FerruleScalar Scalar(FerruleValue value) => new(value);

static void AssertInstance(FerruleInstance actual, FerruleInstance expected, string output)
{
    switch (actual, expected)
    {
        case (FerruleScalar left, FerruleScalar right) when left.Value == right.Value:
            return;
        case (FerruleGroup left, FerruleGroup right) when left.Fields.Count == right.Fields.Count:
            for (var index = 0; index < left.Fields.Count; index++)
            {
                if (left.Fields[index].Name != right.Fields[index].Name)
                {
                    break;
                }
                AssertInstance(left.Fields[index].Value, right.Fields[index].Value, output);
                if (index + 1 == left.Fields.Count)
                {
                    return;
                }
            }
            if (left.Fields.Count == 0)
            {
                return;
            }
            break;
    }

    throw new InvalidOperationException($"Generated {output} output differs from the engine.");
}

enum Case
{
    None,
    Source,
    Generated,
    Named,
    EmptyMessage,
    NonBoolean,
}
