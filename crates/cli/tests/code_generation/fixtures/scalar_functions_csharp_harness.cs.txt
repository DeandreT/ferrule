using Ferrule.Generated;
using Ferrule.Runtime;

var output = (FerruleGroup)GeneratedMapping.Execute(Source(false, false));
foreach (var (group, name, expected) in ExpectedValues())
{
    AssertValue(output, group, name, expected);
}
AssertError(
    () => GeneratedMapping.Execute(Source(true, false)),
    FerruleRuntimeError.FunctionType,
    "normalize_space",
    found: FerruleValueKind.Int64);
AssertError(
    () => GeneratedMapping.Execute(Source(false, true)),
    FerruleRuntimeError.FunctionArity,
    "resolve_filepath",
    expectedArity: 2,
    actualArity: 1);

static FerruleGroup Source(bool failType, bool failArity) => new(new FerruleField[]
{
    new("FailType", new FerruleScalar(FerruleValue.FromBoolean(failType))),
    new("FailArity", new FerruleScalar(FerruleValue.FromBoolean(failArity))),
});

static void AssertValue(FerruleGroup output, string group, string name, FerruleValue expected)
{
    var nested = (FerruleGroup)output.Fields.Single(field => field.Name == group).Value;
    var value = (FerruleScalar)nested.Fields.Single(field => field.Name == name).Value;
    if (value.Value != expected)
    {
        throw new InvalidOperationException($"Scalar function output {group}/{name} differs.");
    }
}

static void AssertError(
    Action action,
    FerruleRuntimeError expected,
    string function,
    int? expectedArity = null,
    int? actualArity = null,
    FerruleValueKind? found = null)
{
    try
    {
        action();
    }
    catch (FerruleRuntimeException error) when (
        error.Error == expected &&
        error.Function == function &&
        error.ExpectedArity == expectedArity &&
        error.ActualArity == actualArity &&
        error.FoundKind == found)
    {
        return;
    }

    throw new InvalidOperationException($"Expected {expected} from {function}.");
}
