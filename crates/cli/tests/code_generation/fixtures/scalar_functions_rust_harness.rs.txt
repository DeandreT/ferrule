use codegen_runtime::{FunctionError, Instance, RuntimeError, Value, field, group, scalar};
use ferrule_generated_mapping::execute;

fn main() -> Result<(), Box<dyn std::error::Error>> {
    let output = execute(&source(false, false))?;
    for (group, name, expected) in expected_values() {
        assert_value(&output, group, name, expected);
    }
    assert_eq!(
        execute(&source(true, false)),
        Err(RuntimeError::Function(FunctionError::TypeMismatch {
            function: "normalize_space",
            got: "int",
        })),
    );
    assert_eq!(
        execute(&source(false, true)),
        Err(RuntimeError::Function(FunctionError::ArityMismatch {
            function: "resolve_filepath",
            expected: 2,
            got: 1,
        })),
    );
    Ok(())
}

fn source(fail_type: bool, fail_arity: bool) -> Instance {
    group([
        field("FailType", scalar(Value::Bool(fail_type))),
        field("FailArity", scalar(Value::Bool(fail_arity))),
    ])
}

fn assert_value(output: &Instance, group: &str, name: &str, expected: Value) {
    let actual = output
        .field(group)
        .and_then(|value| value.field(name))
        .and_then(Instance::as_scalar);
    assert_eq!(actual, Some(&expected), "{group}/{name}");
}
